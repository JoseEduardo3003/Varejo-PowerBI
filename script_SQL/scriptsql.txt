Tabela Produto:
SELECT DISTINCT ProductKey as id_produto,
Product_Name as nome_produto,

Tabela Customers
- Calculando coluna Idade
SELECT 
    Name,
    Birthday,
    DATEDIFF(YEAR, Birthday, GETDATE()) 
        - CASE 
            WHEN MONTH(Birthday) > MONTH(GETDATE()) 
              OR (MONTH(Birthday) = MONTH(GETDATE()) AND DAY(Birthday) > DAY(GETDATE()))
            THEN 1 
            ELSE 0 
          END AS Idade
FROM Customers;

SELECT DISTINCT CustomerKey AS id_cliente,
Gender as genero,

Tabela Stores: SELECT StoreKey AS id_loja,
Country AS pais,
State AS estado,
Square_Meters AS tamanho_loja
FROM Stores

Tabela Exchange_Rates:
SELECT Currency AS id_moeda,
AVG(Exchange) AS taxa_conversao
FROM Exchange_Rates
GROUP BY Currency;

Tabela Sales:
SELECT Order_Number AS id_pedido,
Sales.Order_Date AS data_pedido,
Sales.CustomerKey AS id_cliente,
Sales.StoreKey AS id_loja,
Sales.ProductKey AS id_produto,
Sales.Quantity AS quantidade,
Sales.Currency_Code AS id_moeda,
(Quantity * Products.Unit_Price_USD) AS custo_total
FROM Sales
LEFT JOIN Products ON Sales.ProductKey = Products.ProductKey;
